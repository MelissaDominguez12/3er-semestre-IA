void insertionSort(int arr[]){
    int i, j, key;
    for (int i = 1; i < tam; i++)
    {
        key = arr[i];
        j = i - 1;
        
        while (j >= 0 && arr[j] > key)
        {
            arr[j+1] = arr[j];
            j = j - 1; 
        }
        arr[j+1]= key;
        
    }
     for (int i = 0; i < tam; i++)
    {
        printf("%d) %d\n", i+1, arr[i]);
    }
    
}

void bubbleSort(int arr[]){
    int i, j, k, aux;
    for (i = 0; i < tam; i++)
    {
        for (j = 0; j < tam; j++)
        {
            if (arr[j] > arr[j+1])
            {
                aux = arr[j];
                arr[j] = arr[j+1];
                arr[j+1] = aux;
            }
            
        }
        
    }
    for (k = 0; k < tam; k++)
    {
        printf("%d) %d\n", k+1, arr[k]);
    }
    
    
}
